2. Star Pyramid Pattern

rows = int(input('Enter the rows: '))
for i in range(1, rows+1):
    for j in range(0, i):
        print('*', end=' ')
    print()
for i in range(1, rows):
    for j in range(rows - i, 0, -1):
        print('*', end=' ')
    print()

9. Decimal to Binary

n = int(input('Enter number: '))
binary = []
while n > 0:
    binary.append(n % 2)
    n //= 2
for i in binary[::-1]:
    print(i, end='')

10. LCM and HCF of Two Numbers

x = int(input('Enter 1st number: '))
y = int(input('Enter 2nd number: '))
greater = x if x > y else y
while True:
    if greater % x == 0 and greater % y == 0:
        lcm = greater
        break
    greater += 1
print("LCM:", lcm)

smaller = x if x < y else y
for i in range(1, smaller+1):
    if x % i == 0 and y % i == 0:
        hcf = i
print("HCF:", hcf)

14. Sum and Average of Digits in String

s = 'hel123O4every093'
sum_d = 0
count = 0
for i in s:
    if i.isdigit():
        sum_d += int(i)
        count += 1
print('Sum:', sum_d)
print('Avg:', sum_d / count)
print('Count:', count)

16. Check Symmetrical String

s = input('Enter string: ')
if len(s) % 2 == 0:
    s1 = s[:len(s)//2]
    s2 = s[len(s)//2:]
else:
    s1 = s[:len(s)//2]
    s2 = s[len(s)//2 + 1:]
print("symmetrical" if s1 == s2 else "not symmetrical")

18. Find Uncommon Words from Two Strings

A = "apple banana mango"
B = "banana fruits mango"
L = []
for i in A.split():
    if i not in B and i not in L:
        L.append(i)
for j in B.split():
    if j not in A and j not in L:
        L.append(j)
print(L)
