7. Reverse a Given Number

number = int(input('Enter the number: '))
rev = 0
while number > 0:
    last = number % 10
    rev = rev * 10 + last
    number //= 10
print(rev)

8. Sum from 1 to N (skip multiples of 5, stop if sum ≥ 300)

N = int(input('Enter the number: '))
total = 0
i = 1
while i <= N:
    if i % 5 == 0:
        i += 1
        continue
    total += i
    if total > 300:
        total -= i
        break
    i += 1
print(total)

12. Robot Movement from Origin

pos = [0, 0]
while True:
    s = input('Enter the robot path: ')
    if s == '!':
        break
    direction, steps = s.split()[0], int(s.split()[1])
    if direction == 'UP':
        pos[1] += steps
    elif direction == 'DOWN':
        pos[1] -= steps
    elif direction == 'LEFT':
        pos[0] -= steps
    elif direction == 'RIGHT':
        pos[0] += steps
print('New Position:', pos)
print((pos[0]**2 + pos[1]**2)**0.5)

13. Prime Number Check

num = int(input('Enter a number: '))
if num <= 1:
    print(num, "is not a prime number.")
else:
    is_prime = True
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            is_prime = False
            break
    if is_prime:
        print(num, "is a prime number.")
    else:
        print(num, "is not a prime number.")

14. Armstrong Numbers in a Range

start = int(input("Enter start: "))
end = int(input("Enter end: "))
for num in range(start, end + 1):
    temp = num
    s = 0
    order = len(str(num))
    while temp > 0:
        digit = temp % 10
        s += digit ** order
        temp //= 10
    if num == s:
        print(num)

15. Angle Between Hour and Minute Hands

h = int(input('Enter hours hand: '))
m = int(input('Enter minute hand: '))
if (h < 0 or m < 0 or h > 12 or m > 60):
    print('Wrong input')
else:
    m_angle = m * 6
    h_angle = h * 30 + m * 0.5
    angle = abs(h_angle - m_angle)
    if angle > 180:
        angle = 360 - angle
    print(int(angle))

16. Check if Two Rectangles Overlap

l1_x, l1_y = map(int, input('Enter L1 coords: ').split())
r1_x, r1_y = map(int, input('Enter R1 coords: ').split())
l2_x, l2_y = map(int, input('Enter L2 coords: ').split())
r2_x, r2_y = map(int, input('Enter R2 coords: ').split())

if l1_x == r1_x or l1_y == r1_y or r2_x == l2_x or l2_y == r2_y:
    print("Don't Overlap")
elif l1_x > r2_x or l2_x > r1_x:
    print("Don't Overlap")
elif r1_y > l2_y or r2_y > l1_y:
    print("Don't Overlap")
else:
    print("Overlap")
Example: L1 1 1, R1 4 4, L2 2 2, R2 5 5 → Overlap
