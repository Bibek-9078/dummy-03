Problem 5: Sum of elements greater than or equal to each element

You are given a list of integers. You are asked to make a list by running through elements of the list by adding all elements greater and itself. 
L = [2,4,6,10,1]
result = []

for i in L:
    sum = 0
    for j in L:
        if i <= j:
            sum = sum + j
    result.append(sum)

print(result)
[22, 20, 16, 10, 23]

Problem 7: Sort alphanumeric strings by product of digits

Sort a list of alphanumeric strings based on product value of numeric characters in it. If no numeric character, product value is 1. 
L = ['1ac21', '23fg', '456', '098d', '1', 'kls']
prod_value = []
for item in L:
    product = 1
    for char in item:
        if char.isdigit():
            product *= int(char)
    prod_value.append(product)

sorted_items = [i[1] for i in sorted(list(zip(prod_value, L)), reverse=True)]
print(sorted_items)
['456', '23fg', '1ac21', '1', 'kls', '098d']

Problem 10: Add Space between Potential Words

test_list = ["campusxIs", "bestFor", "dataScientist"]
res = []
for i in test_list:
    temp = [[]]
    for char in i:
        if char.isupper():
            temp.append([])
        temp[-1].append(char)
    temp_string = ""
    for item in temp:
        temp_string = temp_string + "".join(item) + " "
    res.append(temp_string[0:-1])
print(res)
['campusx Is', 'best For', 'data Scientist']
